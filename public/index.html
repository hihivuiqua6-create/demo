Quick Start

// 1. Lấy API key từ Dashboard → API Keys → Tạo key
// 2. Gọi API từ web của bạn:

const response = await fetch('https://YOUR_APP.onrender.com/api/create-bot', {
  method: 'POST',
  headers: {
    'Content-Type': 'application/json',
    'X-Api-Key': 'YOUR_API_KEY'
  },
  body: JSON.stringify({
    token: 'BOT_TOKEN_FROM_BOTFATHER',
    web_url: 'https://yourwebsite.com',
    welcome_message: 'Xin chào! 👋',
    button_text: '🚀 Mở App'
  })
});

const data = await response.json();
